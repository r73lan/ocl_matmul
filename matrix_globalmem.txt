kernel void add(global const float* A, global const float* B, global float* C, uint M, uint N, uint K)
{
    uint x = get_global_id(0);
    uint y = get_global_id(1);
    float sum = 0.0;
    for (uint i = 0; i < K; ++i)
    {
    	sum += A[y * K + i] * B[i * N + x];
    }
    C[y * N + x] = sum;
}
